Run Mongo container instance (port forwarding port on local machine to container default mongo port)
docker run -d -p 27017:27017 --name aspnetrun-mongo mongo
Find running container
docker ps -f name=aspnetrun-mongo
Start container
docker start aspnetrun-mongo
Run bash terminal in container
docker exec -it aspnetrun-mongo /bin/bash
Run aspnet app docker file with docker compose
docker-compose -f docker-compose.yml -f docker-compose.override.yml up -d
Setup volume storage for mongo environment variable
export WEBAPP_STORAGE_HOME=$HOME/storage

export WEBAPP_STORAGE_HOME=$HOME/storage
